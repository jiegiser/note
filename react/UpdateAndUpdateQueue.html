# Update

# 用处

  • 用于记录组件状态的改变
  • 存放于 UpdateQueue 中
  • 多个 Update 可以同时存在

# 源码

// 在 render 中调用 updateContainer 去创建更新；
// 在 updateContainer 中的 createUpdate 创建一个 update

export function updateContainer(
  element: ReactNodeList,
  container: OpaqueRoot,
  parentComponent: ?React$Component<any, any>,
  callback: ?Function,
): Lane {
  const current = container.current;
  // time
  const eventTime = requestEventTime();
  const lane = requestUpdateLane(current);

  if (enableSchedulingProfiler) {
    markRenderScheduled(lane);
  }

  const context = getContextForSubtree(parentComponent);
  if (container.context === null) {
    container.context = context;
  } else {
    container.pendingContext = context;
  }
  // scheduleRootUpdate
  // 标记应用更新节点
  const update = createUpdate(eventTime, lane);
  // Caution: React DevTools currently depends on this property
  // being called "element".
  update.payload = {element};

  callback = callback === undefined ? null : callback;
  if (callback !== null) {
    update.callback = callback;
  }

  // 将 updata 加入到 fiber 对象对应的 updataqueue 里面
  // current 就是 fiber 对象
  enqueueUpdate(current, update, lane);

  // 任务调度
  const root = scheduleUpdateOnFiber(current, lane, eventTime);
  if (root !== null) {
    entangleTransitions(root, current, lane);
  }

  return lane;
}



// createUpdate
export function createUpdate(eventTime: number, lane: Lane): Update<*> {
  const update: Update<*> = {
    // 获取程序运行到目前为止的时间，用于进行任务优先级排序
    eventTime,
    // 优先级区间模型被称为 lanes（车道模型）
    lane,
    // export const UpdateState = 0; 更新
    // export const ReplaceState = 1; 替代
    // export const ForceUpdate = 2; 强制更新
    // export const CaptureUpdate = 3; 捕获错误，生成 update
    // 指定更新的类型，值为以上几种
    tag: UpdateState,
    // 更新内容，比如 setState 接收的第一个参数
    payload: null,
    // 对应的回调， setState ， render 都有
    callback: null,
    // 指向下一个更新
    next: null,
  };
  return update;
}

下面是 UpdateQueue 返回的类型：

export function initializeUpdateQueue<State>(fiber: Fiber): void {
  const queue: UpdateQueue<State> = {
    // 每次操作完更新之后的 state
    baseState: fiber.memoizedState,
    // 队列中的第一个 Update
    firstBaseUpdate: null,
    // 队列中的最后一个 Update
    lastBaseUpdate: null,
    shared: {
      pending: null,
      interleaved: null,
      lanes: NoLanes,
    },
    effects: null,
  };
  fiber.updateQueue = queue;
}

最后通过 enqueueUpdate 将 update 放入 fiber 的 updateQueue：

export function enqueueUpdate<State>(
  fiber: Fiber,
  update: Update<State>,
  lane: Lane,
) {
  const updateQueue = fiber.updateQueue;
  if (updateQueue === null) {
    // Only occurs if the fiber has been unmounted.
    return;
  }

  const sharedQueue: SharedQueue<State> = (updateQueue: any).shared;

  if (isInterleavedUpdate(fiber, lane)) {
    const interleaved = sharedQueue.interleaved;
    if (interleaved === null) {
      // This is the first update. Create a circular list.
      update.next = update;
      // At the end of the current render, this queue's interleaved updates will
      // be transfered to the pending queue.
      pushInterleavedQueue(sharedQueue);
    } else {
      update.next = interleaved.next;
      interleaved.next = update;
    }
    sharedQueue.interleaved = update;
  } else {
    const pending = sharedQueue.pending;
    if (pending === null) {
      // This is the first update. Create a circular list.
      update.next = update;
    } else {
      update.next = pending.next;
      pending.next = update;
    }
    sharedQueue.pending = update;
  }
}
