# 适配器模式

# 介绍

  • 旧接口格式和使用者不兼容
  • 中间加一个适配转换接口

# UML 类图

# 代码演示：

class Adaptee {
  specificRequeest() {
    return '德国标准插头'
  }
}
class Target {
  constructor() {
    this.adaptee = new Adaptee()
  }
  request() {
    let info = this.adaptee.specificRequeest()
    return `${info} - 准换器 - 中国标准插头`
  }
}

// 测试

let target = new Target()
let res = target.request()
console.log(res)
